    public static Connection connect() throws SQLException {
        return DriverManager.getConnection(URL);
    }

    private static int executeUpdate(String sql, Object... params) throws SQLException {
        try (Connection conn = connect();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            bindParams(stmt, params);
            return stmt.executeUpdate();
        }
    }

    private static <T> T executeQuery(String sql, ResultSetHandler<T> handler, Object... params) throws SQLException {
        try (Connection conn = connect();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            bindParams(stmt, params);
            try (ResultSet rs = stmt.executeQuery()) {
                return handler.handle(rs);
            }
        }
    }

    private static void bindParams(PreparedStatement stmt, Object... params) throws SQLException {
        if (params == null) return;
        for (int i = 0; i < params.length; i++) {
            stmt.setObject(i + 1, params[i]);
        }
    }

    @FunctionalInterface
    private interface ResultSetHandler<T> {
        T handle(ResultSet rs) throws SQLException;
    }

    public static boolean createUser(User user) {
        String username = user.getUSERNAME() == null ? "" : user.getUSERNAME().trim();
        String password = user.getPASSWORD() == null ? "" : user.getPASSWORD().trim();
        String category = user.getCATEGORY() == null ? "normal" : user.getCATEGORY();

        if (username.isEmpty() || password.isEmpty()) return false;

        String check = "SELECT 1 FROM users WHERE lower(username)=lower(?)";
        try {
            boolean exists = executeQuery(check, rs -> rs.next(), username);
            if (exists) return false;
        } catch (SQLException e) {
            System.out.println("Invalid");
        }

        String sql = "INSERT INTO users(username,password,category) VALUES(?,?,?)";
        try {
            int rows = executeUpdate(sql, username, password, category);
            return rows > 0;
        } catch (SQLException e) {
            System.out.println("Invalid");
            return false;
        }
    }

    public static boolean updateBookingTimeOutAndTotals(int bookingId, String newTimeOut, Double newTotalAmount, Double paid, Double change) {
        StringBuilder sb = new StringBuilder("UPDATE bookings SET time_out = ?");
        if (newTotalAmount != null) sb.append(", total_amount = ?");
        if (paid != null) sb.append(", paid = ?");
        if (change != null) sb.append(", change_amount = ?");
        sb.append(" WHERE id = ?");
        try {
            if (newTotalAmount == null && paid == null && change == null) {
                int rows = executeUpdate(sb.toString(), newTimeOut, bookingId);
                return rows > 0;
            } else if (paid == null && change == null) {
                int rows = executeUpdate(sb.toString(), newTimeOut, newTotalAmount, bookingId);
                return rows > 0;
            } else if (change == null) {
                int rows = executeUpdate(sb.toString(), newTimeOut, newTotalAmount, paid, bookingId);
                return rows > 0;
            } else {
                int rows = executeUpdate(sb.toString(), newTimeOut, newTotalAmount, paid, change, bookingId);
                return rows > 0;
            }
        } catch (SQLException e) {
            System.out.println("Invalid");
            return false;
        }
    }

            System.out.println("Invalid");

    public static void printProfitReport() {
        String sqlIncome = "SELECT IFNULL(SUM(total_amount), 0.0) as total FROM bookings";
        String sqlExpense = "SELECT IFNULL(SUM(cost), 0.0) as total FROM maintenance";
        String sqlBills = "SELECT IFNULL(SUM(amount), 0.0) as total FROM bills";
        String sqlSalaries = "SELECT IFNULL(SUM(salary), 0.0) as total FROM employees";

        try {
            Double income = executeQuery(sqlIncome, rs -> rs.next() ? rs.getDouble("total") : 0.0);
            Double expense = executeQuery(sqlExpense, rs -> rs.next() ? rs.getDouble("total") : 0.0);
            Double bills = executeQuery(sqlBills, rs -> rs.next() ? rs.getDouble("total") : 0.0);
            Double salaries = executeQuery(sqlSalaries, rs -> rs.next() ? rs.getDouble("total") : 0.0);

            if (income == null) income = 0.0;
            if (expense == null) expense = 0.0;
            if (bills == null) bills = 0.0;
            if (salaries == null) salaries = 0.0;

            double totalExpenses = expense + bills + salaries;
            double grossProfit = income - totalExpenses;

            double tax = 0.0;
            if (grossProfit > 0) {
                tax = roundTwoDecimals(grossProfit * 0.30);
            }
            double netProfit = roundTwoDecimals(grossProfit - tax);

            System.out.printf("Total Income: %.2f%n", income);
            System.out.printf("Maintenance Expenses: %.2f%n", expense);
            System.out.printf("Bills: %.2f%n", bills);
            System.out.printf("Employees salaries (total): %.2f%n", salaries);
            System.out.printf("Total Expenses: %.2f%n", totalExpenses);
            System.out.printf("Gross Profit (before tax): %.2f%n", grossProfit);
            System.out.printf("Tax (30%% if positive): %.2f%n", tax);
            System.out.printf("Net Profit: %.2f%n", netProfit);
        } catch (SQLException e) {
            System.out.println("Invalid");
        }
    }

    public static synchronized void addToCompanyIncome(BigDecimal amount) {
        String select = "SELECT value FROM company_settings WHERE key='total_income'";
        try {
            String current = executeQuery(select, rs -> rs.next() ? rs.getString("value") : null);
            BigDecimal currentVal = (current == null) ? BigDecimal.ZERO : new BigDecimal(current);
            BigDecimal next = currentVal.add(amount).setScale(2, RoundingMode.HALF_UP);
            String upsert = "INSERT OR REPLACE INTO company_settings(key, value) VALUES('total_income', ?)";
            executeUpdate(upsert, next.toPlainString());
        } catch (SQLException e) {
            System.out.println("Invalid");
        }
    }

    public static double getTotalIncomeFromDB() {
        String sql = "SELECT value FROM company_settings WHERE key='total_income'";
        try {
            String val = executeQuery(sql, rs -> rs.next() ? rs.getString("value") : null);
            if (val == null) {
                // fallback to computing from bookings.total_amount
                String sqlBookings = "SELECT IFNULL(SUM(total_amount), 0.0) as total FROM bookings";
                Double income = executeQuery(sqlBookings, rs -> rs.next() ? rs.getDouble("total") : 0.0);
                return (income == null) ? 0.0 : roundTwoDecimals(income);
            }
            return new BigDecimal(val).setScale(2, RoundingMode.HALF_UP).doubleValue();
        } catch (SQLException e) {
            System.out.println("Invalid");
            return 0.0;
        }
    }
    private static double roundTwoDecimals(double v) {
        return Math.round(v * 100.0) / 100.0;
    }
