    private static String readOrBack(Scanner sc) {
        while (true) {
            String s = sc.nextLine();
            if (s == null) return null;
            s = s.trim();
            if (s.equalsIgnoreCase("b")) {
                while (true) {
                    System.out.print("Go back to main menu? (yes/no): ");
                    String conf = sc.nextLine();
                    if (conf == null) return null;
                    conf = conf.trim();
                    if (conf.equalsIgnoreCase("yes")) {
                        return null;
                    }
                    if (conf.equalsIgnoreCase("no")) {
                        System.out.println("Continuing current operation.");
                        break;
                    }
                    System.out.println("Invalid input. Please enter 'yes' or 'no'.");
                }
                continue;
            }
            return s;
        }
    }

    private static double computeHours(String timeIn, String timeOut) {
        LocalTime in = LocalTime.parse(timeIn, DateTimeFormatter.ofPattern("HH:mm"));
        LocalTime out = LocalTime.parse(timeOut, DateTimeFormatter.ofPattern("HH:mm"));
        Duration d = Duration.between(in, out);
        double minutes = d.toMinutes();
        return minutes / 60.0;
    }

    private static double roundTwoDecimals(double v) {
        return Math.round(v * 100.0) / 100.0;
    }

    private static void adminCheckAvailability(Scanner sc) {
        System.out.print("Type (workspace/meeting_room/both): ");
        String type = readOrBack(sc);
        if (type == null) return;
        type = type.trim().toLowerCase();
        if (type.equals("b") || type.equals("both")) type = "both";
        if(!type.equals("workspace") && !type.equals("meeting_room") && !type.equals("both")) {
            System.out.println("Invalid input.");
            return;
        }

        System.out.print("Date (YYYY-MM-DD): ");
        String date = readOrBack(sc);
        if (date == null) return;
        if (!Booking.isValidDate(date)) {
            System.out.println("Invalid date format.");
            return;
        }
        if (isDateBeforeToday(date)) {
            System.out.println("Cannot check availability for past dates. Enter today or a future date.");
            return;
        }

        System.out.print("Time In (HH:mm): ");
        String timeIn = readOrBack(sc);
        if (timeIn == null) return;

        System.out.print("Time Out (HH:mm): ");
        String timeOut = readOrBack(sc);
        if (timeOut == null) return;

        if (!Booking.isValidTime(timeIn) || !Booking.isValidTime(timeOut)) {
            System.out.println("Invalid time.");
            return;
        }

        if (type.equals("both") || type.equals("workspace")) {
            System.out.println("Workspace availability:");
            DatabaseConnection.checkAvailability("workspace", date, timeIn, timeOut);
        }
        if (type.equals("both") || type.equals("meeting_room")) {
            System.out.println("Meeting room availability:");
            DatabaseConnection.checkAvailability("meeting_room", date, timeIn, timeOut);
        }
    }

    private static boolean isDateBeforeToday(String dateStr) {
        try {
            LocalDate d = LocalDate.parse(dateStr, DateTimeFormatter.ISO_LOCAL_DATE);
            return d.isBefore(LocalDate.now());
        } catch (DateTimeParseException e) {
            return true;
        }
    }
    private static void addHoursFlow(Scanner sc, String currentUser) {
        System.out.print("Enter Booking ID to add hours: ");
        String idStr = readOrBack(sc);
        if (idStr == null) return;
        int bookingId;
        try {
            bookingId = Integer.parseInt(idStr.trim());
        } catch (NumberFormatException e) {
            System.out.println("Invalid booking id.");
            return;
        }

        Booking b = DatabaseConnection.getBookingById(bookingId);
        if (b == null) {
            System.out.println("Booking not found.");
            return;
        }
        if (!b.getUSERNAME().equals(currentUser)) {
            System.out.println("This booking does not belong to you.");
            return;
        }
        if (!"reserved".equalsIgnoreCase(b.getSTATUS())) {
            System.out.println("Only active reserved bookings can be extended.");
            return;
        }

        String date = b.getDATE();
        String type = b.getTYPE();
        int room = b.getROOMNUMBER();
        String currentOut = b.getTIMEOUT(); // "HH:mm"
        if (currentOut == null || currentOut.isEmpty()) {
            System.out.println("Current booking has no time out set.");
            return;
        }

        String nextStart = DatabaseConnection.getNextBookingStart(type, date, room, currentOut, bookingId);

        LocalTime curOutTime;
        try {
            curOutTime = LocalTime.parse(currentOut, DateTimeFormatter.ofPattern("HH:mm"));
        } catch (Exception ex) {
            System.out.println("Invalid stored time_out format.");
            return;
        }

        LocalTime limitTime;
        if (nextStart != null) {
            try {
                limitTime = LocalTime.parse(nextStart, DateTimeFormatter.ofPattern("HH:mm"));
            } catch (Exception ex) {
                System.out.println("Invalid next booking time format.");
                return;
            }
        } else {
            limitTime = LocalTime.of(23, 59);
        }

        if (!limitTime.isAfter(curOutTime)) {
            System.out.println("No available time to extend for this booking.");
            return;
        }

        long availableMinutes = Duration.between(curOutTime, limitTime).toMinutes();
        int maxHours = (int) (availableMinutes / 60); // whole hours only
        if (maxHours <= 0) {
            System.out.println("Available extension less than 1 hour (" + availableMinutes + " minutes). Cannot add hours.");
            return;
        }

        System.out.println("You can add up to " + maxHours + " hour(s). Current time out: " + currentOut + (nextStart != null ? " (next booking at " + nextStart + ")" : " (no next booking)"));
        System.out.print("Enter number of hours to add (1-" + maxHours + "): ");
        String hrsStr = readOrBack(sc);
        if (hrsStr == null) return;
        int addHours;
        try {
            addHours = Integer.parseInt(hrsStr.trim());
        } catch (NumberFormatException e) {
            System.out.println("Invalid number.");
            return;
        }
        if (addHours < 1 || addHours > maxHours) {
            System.out.println("Invalid hours. Must be between 1 and " + maxHours);
            return;
        }

        LocalTime newOutTime = curOutTime.plusHours(addHours);
        String newOut = newOutTime.format(DateTimeFormatter.ofPattern("HH:mm"));

        double pricePerHour = DatabaseConnection.getPricePerHour();
        double extraCharge = roundTwoDecimals(addHours * pricePerHour);

        System.out.printf("Extra charge for %d hour(s): %.2f%n", addHours, extraCharge);
        System.out.println("Payment methods: 1 Card  2 Cash");
        String pm;
        while (true) {
            System.out.print("Choose payment method (1 or 2): ");
            pm = readOrBack(sc);
            if (pm == null) return;
            pm = pm.trim();
            if (pm.equals("1") || pm.equals("2")) break;
            System.out.println("Invalid input.");
        }

        if (pm.equals("1")) {
            System.out.print("Enter card number: ");
            String card = readOrBack(sc);
            if (card == null) return;
            PaymentFramework payment = new CardPayment(extraCharge, 0.0, card);
            payment.processInvoice();
            if (!payment.validatePayment()) {
                System.out.println("Card validation failed. Extension not accepted.");
                return;
            }
            double newTotal = roundTwoDecimals(b.getTOTALAMOUNT() + extraCharge);
            boolean ok = DatabaseConnection.updateBookingTimeOutAndTotals(bookingId, newOut, newTotal, null, null);
            if (!ok) {
                System.out.println("Failed to update booking.");
                return;
            }
            DatabaseConnection.recordPayment(currentUser, type, date, b.getTIMEIN(), newOut, room, extraCharge, 0.0, extraCharge);
            System.out.println("Extension successful. New time out: " + newOut);
        } else {
            while (true) {
                System.out.print("Enter payment amount: ");
                String paidStr = readOrBack(sc);
                if (paidStr == null) return;
                try {
                    double paid = Double.parseDouble(paidStr.trim());
                    if (paid < extraCharge) {
                        System.out.println("Insufficient payment. Please enter an amount >= " + extraCharge);
                        continue;
                    }
                    double change = roundTwoDecimals(paid - extraCharge);
                    double newTotal = roundTwoDecimals(b.getTOTALAMOUNT() + extraCharge);
                    boolean ok = DatabaseConnection.updateBookingTimeOutAndTotals(bookingId, newOut, newTotal, paid, change);
                    if (!ok) {
                        System.out.println("Failed to update booking.");
                        return;
                    }
                    DatabaseConnection.recordPayment(currentUser, type, date, b.getTIMEIN(), newOut, room, paid, change, extraCharge);
                    System.out.println("Extension successful. New time out: " + newOut);
                    if (change > 0) System.out.println("Change: " + change);
                    break;
                } catch (NumberFormatException ex) {
                    System.out.println("Invalid number.");
                }
            }
        }
    }
