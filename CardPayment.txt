import java.math.BigDecimal;
import java.math.RoundingMode;

public abstract class PaymentFramework {

    protected double amount;
    protected double discount;
    protected static final double vatRate = 0.12;

    private static double totalCompanyIncome = 0;

    public PaymentFramework(double amount, double discount) {
        if (amount < 0) throw new IllegalArgumentException("amount must be >= 0");
        if (discount < 0.0 || discount > 1.0) throw new IllegalArgumentException("discount must be between 0.0 and 1.0");
        this.amount = amount;
        this.discount = discount;
    }

    public abstract boolean validatePayment();

    public double applyDiscount() {
        BigDecimal amt = BigDecimal.valueOf(amount);
        BigDecimal disc = amt.multiply(BigDecimal.valueOf(discount));
        BigDecimal after = amt.subtract(disc).setScale(2, RoundingMode.HALF_UP);
        return after.doubleValue();
    }

    public double applyVAT(double amountAfterDiscount) {
        return BigDecimal.valueOf(amountAfterDiscount).setScale(2, RoundingMode.HALF_UP).doubleValue();
    }

    public void finalizeTransaction(double finalAmount) {
        System.out.println("Transaction successful!");
        System.out.println("Total amount paid: " + finalAmount);

        synchronized (PaymentFramework.class) {
            BigDecimal current = BigDecimal.valueOf(totalCompanyIncome);
            BigDecimal next = current.add(BigDecimal.valueOf(finalAmount)).setScale(2, RoundingMode.HALF_UP);
            totalCompanyIncome = next.doubleValue();
        }

        try {
            DatabaseConnection.addToCompanyIncome(java.math.BigDecimal.valueOf(finalAmount));
        } catch (Throwable t) {
            // ignore persistence errors
        }
    }

    public void processInvoice() {
        if (!validatePayment()) {
            System.out.println("Payment validation failed.");
            return;
        }
        double discounted = applyDiscount();
        double finalAmount = applyVAT(discounted);
        finalizeTransaction(finalAmount);
    }

    public static double getTotalCompanyIncome() {
        return totalCompanyIncome;
    }
}
